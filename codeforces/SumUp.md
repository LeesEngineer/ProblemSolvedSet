#include <bits/stdc++.h>

using namespace std;

const int N = 110;

int a[N];

int main()
{
    int t;
    cin >> t;
    while(t --)
    {
        int n;
        cin >> n;
        cin >> a[0];

        int max = 0;
        int cnt = 0;
        for(int i = 1; i < n; i ++)
        {
            cin >> a[i];
            if(a[max] < a[i]) max = i;
            if(a[i] == a[0]) cnt ++;
        }

        if(cnt == n - 1) cout << "No" << endl;
        else
        {
            cout << "Yes" << endl;
            for(int i = 0; i < n; i ++)
            {
                if(i != max) cout << 1 << " ";
                else cout << 2 << " ";
            }
            cout << endl;
        }
    }
}
